 55 /** The STRING LENGTH block definition */
 56 Blockly.Blocks['string_length'] = {
 57   init: function () {
 58     this.appendValueInput('VALUE')
 59       .setCheck('String')
 60       .appendField('longueur de');
 61     this.setOutput(true, 'Number');
 62     this.setColour(160);
 63     this.setTooltip('Returns number of letters in the provided text.');
 64     this.setHelpUrl('http://www.w3schools.com/jsref/jsref_length_string.asp');
 65   }
 66 };
 67 
 68 /* The STRING LENGTH block associated method */
 69 Blockly.JavaScript['string_length'] = function (block) {
 70   var argument0 = Blockly.JavaScript.valueToCode(block, 'VALUE',
 71     Blockly.JavaScript.ORDER_FUNCTION_CALL) || '\'\'';
 72   return [argument0 + '.length', Blockly.JavaScript.ORDER_MEMBER];
 73 };

 92 /** The DISPLAY IMAGE associated method */
 93 Blockly.JavaScript['display_image'] = function (block) {
 94   var argument0 = Blockly.JavaScript.valueToCode(block, 'VALUE',
 95     Blockly.JavaScript.ORDER_FUNCTION_CALL) || '\'\'';
 96   var instruction = "instructionDisplayImage(" + argument0 + ");";
 97   return instruction;
 98 };

100 /**
101  * A COMMENTER
102  *
103  * @param {String} argument0
104  */
105 function instructionDisplayImage(argument0) {
106   var img = document.createElement("img");
107   img.setAttribute("src", argument0);
108   var body = document.getElementById("display");
109   body.style.padding = 0;
110   while (body.hasChildNodes()) {
111     body.removeChild(body.lastChild);
112   }
113   body.appendChild(img);
114 }

166 /** Initialise the method that change the Blockly code to Javascript code and execute it */
167 $(document).ready(function () {
168 
169   var workspace = Blockly.inject('blocklyDiv',
170     { toolbox: document.getElementById('toolbox') });
171 
172   var buttonStart = document.getElementById("button-start");
173   var code;
174   buttonStart.onclick = function () {
175     code = Blockly.JavaScript.workspaceToCode(workspace)
176     console.log(code)
177     try { eval(code) } catch (e) { alert(e) };
178   };
179 });
